% Rules


female(X) :-
     \+male(X).

husband(X, Y) :-  
    male(X),       
    spouse(X, Y).

father(X, Y) :-  
    male(X),       
    parents(Y,X,_);
    parents(Y,_,X).

grandparent(X, Z) :-  
    parents(X, Y),  
    parents(Y, Z).
